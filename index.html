async function sendToDify(text) { const sendBtn = document.getElementById('sendBtn'); sendBtn.disabled = true; try { const res = await fetch('https://digitalerapioneer.com/wp-json/bousai/v1/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ query: text }) }); const data = await res.json(); if (!res.ok) { addMessage('bot', JSON.stringify(data, null, 2)); // エラー内容を表示 return; } addMessage('bot', data.answer || '回答が空です'); } catch (e) { addMessage('bot', '通信エラー: ' + e.message); } finally { sendBtn.disabled = false; } }
